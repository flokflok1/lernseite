```typescript
interface Props {
  window: {
    id: string
    title: string
    icon: string
    payload: {
      courseId: string        // Pflicht
      chapterId?: string      // Optional (nur beim Bearbeiten)
      chapterData?: {         // Optional (vorausgefüllte Daten)
        title: string
        description: string
      }
    }
  }
}
```

### 3.2 isEditMode berechnen

```typescript
const isEditMode = computed(() => !!props.window.payload.chapterId)
```

### 3.3 onMounted - Daten laden

```typescript
onMounted(async () => {
  if (props.window.payload.chapterData) {
    // Vorausgefüllte Daten verwenden
    form.value.title = props.window.payload.chapterData.title || ''
    form.value.description = props.window.payload.chapterData.description || ''
  } else if (props.window.payload.chapterId) {
    // Vom Backend laden
    try {
      const response = await adminGetChapter(props.window.payload.chapterId)
      form.value.title = response.data.chapter.title || ''
      form.value.description = response.data.chapter.description || ''
    } catch (error) {
      console.error('Fehler beim Laden:', error)
    }
  }
})
```

### 3.4 saveChapter - mit Event

```typescript
const saveChapter = async () => {
  if (!form.value.title.trim()) {
    alert('Bitte gib einen Titel ein')
    return
  }

  isSaving.value = true

  try {
    const data = {
      title: form.value.title.trim(),
      description: form.value.description.trim()
    }

    if (isEditMode.value) {
      // Bearbeiten
      await adminUpdateChapter(
        props.window.payload.courseId,
        props.window.payload.chapterId!,
        data
      )
    } else {
      // Neu erstellen
      await adminCreateChapter(props.window.payload.courseId, data)
    }

    // Event für Parent-Page
    window.dispatchEvent(new CustomEvent('chapter-updated'))

    // Window schließen
    windowStore.closeWindow(props.window.id)

  } catch (error) {
    console.error('Fehler beim Speichern:', error)
    alert('Fehler beim Speichern')
  } finally {
    isSaving.value = false
  }
}
```

## PHASE 4: admin.api.ts - Prüfe Funktionen

### 4.1 Stelle sicher dass diese existieren:

```typescript
export const adminGetCourseChapters = (courseId: string) => 
  api.get(`/admin/courses/${courseId}/chapters`)

export const adminGetChapter = (chapterId: string) => 
  api.get(`/admin/chapters/${chapterId}`)

export const adminCreateChapter = (courseId: string, data: {
  title: string
  description?: string
}) => api.post(`/admin/courses/${courseId}/chapters`, data)

export const adminUpdateChapter = (courseId: string, chapterId: string, data: {
  title?: string
  description?: string
}) => api.patch(`/admin/chapters/${chapterId}`, data)

export const adminDeleteChapter = (courseId: string, chapterId: string) => 
  api.delete(`/admin/chapters/${chapterId}`)
```
