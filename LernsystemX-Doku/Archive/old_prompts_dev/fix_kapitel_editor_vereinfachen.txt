### 2.3 onMounted - Daten laden

```typescript
onMounted(async () => {
  // Wenn chapterData übergeben wurde, nutze das
  if (props.window.payload.chapterData) {
    form.value.title = props.window.payload.chapterData.title || ''
    form.value.description = props.window.payload.chapterData.description || ''
  }
  // Sonst lade vom Backend (falls chapterId vorhanden)
  else if (props.window.payload.chapterId) {
    try {
      const response = await adminGetChapter(props.window.payload.chapterId)
      form.value.title = response.data.chapter.title || ''
      form.value.description = response.data.chapter.description || ''
    } catch (error) {
      console.error('Fehler beim Laden des Kapitels:', error)
    }
  }
})
```

### 2.4 saveChapter Funktion - KORREKT

```typescript
const saveChapter = async () => {
  if (!form.value.title.trim()) {
    alert('Bitte gib einen Titel ein')
    return
  }

  isSaving.value = true

  try {
    const data = {
      title: form.value.title.trim(),
      description: form.value.description.trim()
    }

    if (isEditMode.value) {
      // BEARBEITEN
      await adminUpdateChapter(
        props.window.payload.courseId,
        props.window.payload.chapterId!,
        data
      )
      console.log('Kapitel aktualisiert')
    } else {
      // NEU ERSTELLEN
      await adminCreateChapter(
        props.window.payload.courseId,
        data
      )
      console.log('Kapitel erstellt')
    }

    // Window schließen
    windowStore.closeWindow(props.window.id)

    // Optional: Parent-Page neu laden
    window.dispatchEvent(new CustomEvent('chapter-updated'))

  } catch (error) {
    console.error('Fehler beim Speichern:', error)
    alert('Fehler beim Speichern des Kapitels')
  } finally {
    isSaving.value = false
  }
}
```

## PHASE 4: Backend API prüfen

### 4.1 Stelle sicher dass diese Routes existieren:

```python
# backend/app/api/admin_courses.py

@admin_courses_bp.route('/admin/courses/<course_id>/chapters', methods=['POST'])
@jwt_required()
@require_role(['admin'])
def create_chapter(course_id):
    """Erstelle neues Kapitel"""
    try:
        data = request.get_json()

        from app.repositories.chapter_repository import ChapterRepository

        # Automatische order_index berechnen
        existing_chapters = ChapterRepository.find_by_course_id(course_id)
        next_order = len(existing_chapters) + 1

        chapter_data = {
            'course_id': course_id,
            'title': data.get('title'),
            'description': data.get('description', ''),
            'order_index': next_order
        }

        chapter = ChapterRepository.create(chapter_data)

        return jsonify({
            'success': True,
            'data': {'chapter': chapter.to_dict()}
        }), 201
    except Exception as e:
        print(f"Error creating chapter: {e}")
        import traceback
        traceback.print_exc()
        return jsonify({'error': str(e)}), 500


@admin_courses_bp.route('/admin/courses/<course_id>/chapters/<chapter_id>', methods=['PATCH'])
@jwt_required()
@require_role(['admin'])
def update_chapter(course_id, chapter_id):
    """Aktualisiere Kapitel"""
    try:
        data = request.get_json()

        from app.repositories.chapter_repository import ChapterRepository

        chapter = ChapterRepository.update(chapter_id, {
            'title': data.get('title'),
            'description': data.get('description')
        })

        return jsonify({
            'success': True,
            'data': {'chapter': chapter.to_dict()}
        }), 200
    except Exception as e:
        print(f"Error updating chapter: {e}")
        return jsonify({'error': str(e)}), 500
```
