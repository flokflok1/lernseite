## Ziel

1. ❌ "LM00", "LM01" raus → ✅ "00", "01" (ohne LM-Präfix)
2. ❌ Englische Namen raus → ✅ Alles auf Deutsch
3. ❌ Generisches Edit-Formular → ✅ JEDE der 32 Lernmethoden bekommt EIGENES Formular

```vue
<script setup lang="ts">
import { ref, computed, onMounted } from 'vue'
import { useWindowStore } from '@/store/window.store'
import { getLearningMethodByCode } from '@/config/learningMethods'
import { 
  adminCreateLearningMethodInstance,
  adminUpdateLearningMethodInstance 
} from '@/api/admin.api'
import LsxDesktopWindow from '../../LsxDesktopWindow.vue'

interface Props {
  window: {
    id: string
    title: string
    icon: string
    payload: {
      chapterId: string
      methodCode: number
      instanceId?: string      // Wenn gesetzt = Edit-Mode
      instanceData?: any        // Vorausgefüllte Daten
    }
  }
}

const props = defineProps<Props>()
const windowStore = useWindowStore()

const method = computed(() => getLearningMethodByCode(props.window.payload.methodCode)!)
const isEditMode = computed(() => !!props.window.payload.instanceId)
const isSaving = ref(false)

const form = ref({
  title: '',
  instructions: '',
  duration: 15,
  difficulty: 'mittel',
  // METHODEN-SPEZIFISCHE FELDER HIER
})

const windowProps = computed(() => ({
  window: props.window
}))

onMounted(() => {
  if (props.window.payload.instanceData) {
    // Edit-Mode: Daten laden
    form.value.title = props.window.payload.instanceData.title || ''
    form.value.instructions = props.window.payload.instanceData.config?.instructions || ''
    form.value.duration = props.window.payload.instanceData.config?.duration || 15
    form.value.difficulty = props.window.payload.instanceData.config?.difficulty || 'mittel'
    // METHODEN-SPEZIFISCHE FELDER LADEN
  }
})

const save = async () => {
  if (!form.value.title.trim()) {
    alert('Bitte gib einen Titel ein')
    return
  }

  isSaving.value = true

  try {
    const data = {
      methodCode: props.window.payload.methodCode,
      title: form.value.title.trim(),
      config: {
        instructions: form.value.instructions,
        duration: form.value.duration,
        difficulty: form.value.difficulty,
        // METHODEN-SPEZIFISCHE DATEN
      }
    }

    if (isEditMode.value) {
      await adminUpdateLearningMethodInstance(props.window.payload.instanceId!, data)
    } else {
      await adminCreateLearningMethodInstance(props.window.payload.chapterId, data)
    }

    window.dispatchEvent(new CustomEvent('learning-method-updated'))
    close()
  } catch (error) {
    console.error('Fehler beim Speichern:', error)
    alert('Fehler beim Speichern')
  } finally {
    isSaving.value = false
  }
}

const close = () => {
  windowStore.closeWindow(props.window.id)
}
</script>
```

```typescript
const getWindowComponent = (type: WindowType) => {
  // Lernmethoden-Formulare dynamisch laden
  if (type.startsWith('learning-method-') && type.endsWith('-form')) {
    const code = type.replace('learning-method-', '').replace('-form', '')
    const paddedCode = code.padStart(2, '0')
    return defineAsyncComponent(() => 
      import(`./windows/learning-methods/LearningMethod${paddedCode}Form.vue`)
    )
  }

  switch (type) {
    case 'admin-kapitel-editor':
      return defineAsyncComponent(() => import('./windows/AdminKapitelEditorWindow.vue'))
    case 'admin-learning-method-editor':
      return defineAsyncComponent(() => import('./windows/AdminLearningMethodEditorWindow.vue'))
    // ...
  }
}
```
