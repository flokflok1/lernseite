# LERNMETHODEN: Tab mit 4 Gruppen + Erstellen-Buttons (DEUTSCH)

## Ziel

Tab "Lernmethoden" im AdminKapitelEditorWindow mit:
- 4 aufklappbare Gruppen (Accordion)
- 32 Lernmethoden (LM00-LM31) auf DEUTSCH
- "Erstellen" Button pro Methode
- Mehrfach verwendbar (z.B. Quiz 2x im selben Kapitel)

### 2.4 Funktion zum Öffnen des Editors

```typescript
const openLearningMethodEditor = () => {
  if (!props.window.payload.chapterId) return

  windowStore.openWindow({
    type: 'admin-learning-method-editor',
    title: `Lernmethoden: ${props.window.payload.courseTitle}`,
    icon: '🎓',
    payload: {
      courseId: props.window.payload.courseId,
      courseTitle: props.window.payload.courseTitle,
      chapterId: props.window.payload.chapterId,
      chapterTitle: form.value.title
    }
  })
}
```

```vue
<script setup lang="ts">
import { ref, computed, onMounted } from 'vue'
import { useWindowStore } from '@/store/window.store'
import { LEARNING_METHODS, LEARNING_METHOD_CATEGORIES, getLearningMethodsByCategory } from '@/config/learningMethods'
import LsxDesktopWindow from '../LsxDesktopWindow.vue'

interface Props {
  window: {
    id: string
    title: string
    icon: string
    payload: {
      courseId: string
      courseTitle: string
      chapterId: string
      chapterTitle: string
    }
  }
}

const props = defineProps<Props>()
const windowStore = useWindowStore()

const expandedCategories = ref(['explaining']) // Erste Gruppe aufgeklappt
const selectedMethod = ref<any>(null)
const instances = ref<any[]>([])

const windowProps = computed(() => ({
  window: props.window
}))

const toggleCategory = (category: string) => {
  const index = expandedCategories.value.indexOf(category)
  if (index > -1) {
    expandedCategories.value.splice(index, 1)
  } else {
    expandedCategories.value.push(category)
  }
}

const getCategoryIcon = (category: string) => {
  return LEARNING_METHOD_CATEGORIES[category as keyof typeof LEARNING_METHOD_CATEGORIES].icon
}
```
