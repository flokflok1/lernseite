```python
"""Chapter Model"""
from app.extensions import db
from datetime import datetime
import uuid

class Chapter(db.Model):
    __tablename__ = 'chapters'

    chapter_id = db.Column(db.String(36), primary_key=True, default=lambda: str(uuid.uuid4()))
    course_id = db.Column(db.String(36), db.ForeignKey('courses.course_id'), nullable=False)
    title = db.Column(db.String(255), nullable=False)
    description = db.Column(db.Text)
    order_index = db.Column(db.Integer, nullable=False)
    estimated_duration = db.Column(db.Integer)
    created_at = db.Column(db.DateTime, default=datetime.utcnow)
    updated_at = db.Column(db.DateTime, default=datetime.utcnow, onupdate=datetime.utcnow)

    course = db.relationship('Course', back_populates='chapters')
    lessons = db.relationship('Lesson', back_populates='chapter', cascade='all, delete-orphan')
    learning_methods = db.relationship('LearningMethod', back_populates='chapter', cascade='all, delete-orphan')

    def to_dict(self):
        return {
            'chapterId': self.chapter_id,
            'courseId': self.course_id,
            'title': self.title,
            'description': self.description,
            'orderIndex': self.order_index,
            'estimatedDuration': self.estimated_duration,
            'createdAt': self.created_at.isoformat() if self.created_at else None,
            'updatedAt': self.updated_at.isoformat() if self.updated_at else None,
            'lessonCount': len(self.lessons) if self.lessons else 0
        }

    def __repr__(self):
        return f'<Chapter {self.title}>'
```

```python
"""Chapter Repository"""
from app.models.chapter import Chapter
from app.extensions import db
from datetime import datetime
import uuid

class ChapterRepository:

    @staticmethod
    def find_by_course_id(course_id: str):
        """Hole alle Kapitel eines Kurses"""
        return Chapter.query.filter_by(course_id=course_id).order_by(Chapter.order_index).all()

    @staticmethod
    def find_by_id(chapter_id: str):
        """Hole einzelnes Kapitel"""
        return Chapter.query.get(chapter_id)

    @staticmethod
    def create(data: dict):
        """Erstelle neues Kapitel"""
        chapter = Chapter(
            chapter_id=str(uuid.uuid4()),
            course_id=data['courseId'],
            title=data['title'],
            description=data.get('description'),
            order_index=data['orderIndex'],
            estimated_duration=data.get('estimatedDuration'),
            created_at=datetime.utcnow(),
            updated_at=datetime.utcnow()
        )
        db.session.add(chapter)
        db.session.commit()
        return chapter

    @staticmethod
    def update(chapter_id: str, data: dict):
        """Aktualisiere Kapitel"""
        chapter = Chapter.query.get(chapter_id)
        if not chapter:
            raise ValueError('Chapter not found')

        if 'title' in data:
            chapter.title = data['title']
        if 'description' in data:
            chapter.description = data['description']
        if 'orderIndex' in data:
            chapter.order_index = data['orderIndex']
        if 'estimatedDuration' in data:
            chapter.estimated_duration = data['estimatedDuration']

        chapter.updated_at = datetime.utcnow()
        db.session.commit()
        return chapter

    @staticmethod
    def delete(chapter_id: str):
        """Lösche Kapitel (CASCADE löscht lessons + learning_methods)"""
        chapter = Chapter.query.get(chapter_id)
        if chapter:
            db.session.delete(chapter)
            db.session.commit()
            return True
        return False

    @staticmethod
    def get_count_by_course(course_id: str):
        """Anzahl Kapitel in einem Kurs"""
        return Chapter.query.filter_by(course_id=course_id).count()
```

```python
"""Admin Chapters API"""
from flask import Blueprint, request, jsonify
from flask_jwt_extended import jwt_required
from app.repositories.chapter_repository import ChapterRepository
from app.decorators import require_role

admin_chapters_bp = Blueprint('admin_chapters', __name__)

@admin_chapters_bp.route('/admin/chapters/course/<course_id>', methods=['GET'])
@jwt_required()
@require_role(['admin'])
def get_chapters_by_course(course_id):
    """Hole alle Kapitel eines Kurses"""
    try:
        chapters = ChapterRepository.find_by_course_id(course_id)
        return jsonify({
            'success': True,
            'data': {'chapters': [c.to_dict() for c in chapters]}
        }), 200
    except Exception as e:
        return jsonify({'error': str(e)}), 500

@admin_chapters_bp.route('/admin/chapters/<chapter_id>', methods=['GET'])
@jwt_required()
@require_role(['admin'])
def get_chapter(chapter_id):
    """Hole einzelnes Kapitel"""
    try:
        chapter = ChapterRepository.find_by_id(chapter_id)
        if not chapter:
            return jsonify({'error': 'Kapitel nicht gefunden'}), 404
        return jsonify({
            'success': True,
            'data': {'chapter': chapter.to_dict()}
        }), 200
    except Exception as e:
        return jsonify({'error': str(e)}), 500

@admin_chapters_bp.route('/admin/chapters', methods=['POST'])
@jwt_required()
@require_role(['admin'])
def create_chapter():
    """Erstelle neues Kapitel"""
    try:
        data = request.get_json()
        chapter = ChapterRepository.create(data)
        return jsonify({
            'success': True,
            'data': {'chapter': chapter.to_dict()}
        }), 201
    except Exception as e:
        return jsonify({'error': str(e)}), 400

@admin_chapters_bp.route('/admin/chapters/<chapter_id>', methods=['PATCH'])
@jwt_required()
@require_role(['admin'])
def update_chapter(chapter_id):
    """Aktualisiere Kapitel"""
    try:
        data = request.get_json()
        chapter = ChapterRepository.update(chapter_id, data)
        return jsonify({
            'success': True,
            'data': {'chapter': chapter.to_dict()}
        }), 200
    except Exception as e:
        return jsonify({'error': str(e)}), 400

@admin_chapters_bp.route('/admin/chapters/<chapter_id>', methods=['DELETE'])
@jwt_required()
@require_role(['admin'])
def delete_chapter(chapter_id):
    """Lösche Kapitel"""
    try:
        success = ChapterRepository.delete(chapter_id)
        if not success:
            return jsonify({'error': 'Kapitel nicht gefunden'}), 404
        return jsonify({'success': True}), 200
    except Exception as e:
        return jsonify({'error': str(e)}), 500
```

```vue
<script setup lang="ts">
import { ref, onMounted } from 'vue'
import { useWindowStore } from '@/store/window.store'
import { adminGetChapter, adminCreateChapter, adminUpdateChapter } from '@/api/admin.api'

const props = defineProps<{
  windowId: string
  courseId: string
  chapterId?: string
}>()

const windowStore = useWindowStore()
const form = ref({
  title: '',
  description: '',
  orderIndex: 1,
  estimatedDuration: null as number | null
})
const activeTab = ref('info')
const isSaving = ref(false)
const isGenerating = ref(false)
let currentChapterId = props.chapterId

onMounted(async () => {
  if (props.chapterId) {
    try {
      const response = await adminGetChapter(props.chapterId)
      form.value = {
        title: response.data.chapter.title || '',
        description: response.data.chapter.description || '',
        orderIndex: response.data.chapter.orderIndex || 1,
        estimatedDuration: response.data.chapter.estimatedDuration || null
      }
    } catch (error) {
      console.error('Fehler beim Laden des Kapitels:', error)
    }
  }
})

const openAIKapitelGenerator = () => {
  windowStore.openWindow({
    type: 'admin-ai-kapitel-generator',
    props: {
      courseId: props.courseId,
      chapterId: currentChapterId,
      onGenerate: handleAIResult
    }
  })
}

const handleAIResult = (result: any) => {
  form.value.title = result.title
  form.value.description = result.description
  form.value.estimatedDuration = result.estimatedDuration
}

const saveChapter = async () => {
  if (!form.value.title) return

  isSaving.value = true
  try {
    const data = {
      courseId: props.courseId,
      title: form.value.title,
      description: form.value.description,
      orderIndex: form.value.orderIndex,
      estimatedDuration: form.value.estimatedDuration
    }

    if (currentChapterId) {
      await adminUpdateChapter(currentChapterId, data)
    } else {
      const response = await adminCreateChapter(data)
      currentChapterId = response.data.chapter.chapterId
    }

    close()
  } catch (error) {
    console.error('Fehler beim Speichern:', error)
  } finally {
    isSaving.value = false
  }
}

const saveAndEnableTab = async () => {
  await saveChapter()
  if (currentChapterId) {
    activeTab.value = 'lernmethoden'
  }
}

const openLernmethodenEditor = () => {
  windowStore.openWindow({
    type: 'admin-learning-method-editor',
    props: {
      courseId: props.courseId,
      chapterId: currentChapterId
    }
  })
}

const close = () => {
  windowStore.closeWindow(props.windowId)
}
</script>
```

```vue
<script setup lang="ts">
import { ref, onMounted } from 'vue'
import { adminGetCourseFiles, adminGenerateKapitelWithAI } from '@/api/admin.api'
import { useWindowStore } from '@/store/window.store'

const props = defineProps<{
  windowId: string
  courseId: string
  chapterId?: string
  onGenerate: (result: any) => void
}>()

const windowStore = useWindowStore()
const selectedPrompt = ref('kapitel-standard')
const selectedModel = ref('claude-3-5-sonnet')
const courseFiles = ref<any[]>([])
const isGenerating = ref(false)

onMounted(async () => {
  try {
    const response = await adminGetCourseFiles(props.courseId)
    courseFiles.value = response.data.files || []
  } catch (error) {
    console.error('Fehler beim Laden der Dateien:', error)
    courseFiles.value = []
  }
})

const generate = async () => {
  isGenerating.value = true

  try {
    const result = await adminGenerateKapitelWithAI({
      courseId: props.courseId,
      chapterId: props.chapterId,
      promptKey: selectedPrompt.value,
      model: selectedModel.value
    })

    props.onGenerate(result.data)
    close()
  } catch (error) {
    console.error('KI-Generierung fehlgeschlagen:', error)
  } finally {
    isGenerating.value = false
  }
}

const close = () => {
  windowStore.closeWindow(props.windowId)
}

const formatFileSize = (bytes: number) => {
  if (bytes < 1024) return bytes + ' B'
  if (bytes < 1024 * 1024) return (bytes / 1024).toFixed(2) + ' KB'
  return (bytes / (1024 * 1024)).toFixed(2) + ' MB'
}
</script>
```
