**Beispiel: LearningMethod08Form.vue (Karteikarten)**

```vue
<template>
  <BaseLearningMethodForm
    :window="window"
    :method-code="8"
    @save="handleSave"
  >
    <!-- METHODE 08 SPEZIFISCH: NUR Karteikarten -->
    <div class="space-y-4">
      <div class="flex items-center justify-between">
        <label class="block text-sm font-medium">Karteikarten</label>
        <button 
          @click="addCard"
          class="px-3 py-1 text-sm border border-blue-500 text-blue-500 rounded hover:bg-blue-50"
        >
          + Karte hinzufügen
        </button>
      </div>

      <div class="space-y-3 max-h-96 overflow-y-auto">
        <div 
          v-for="(card, index) in formData.cards" 
          :key="index"
          class="p-4 border rounded-lg"
          style="border-color: var(--color-border); background-color: var(--color-surface);"
        >
          <div class="flex items-center justify-between mb-3">
            <span class="text-sm font-medium">Karte {{ index + 1 }}</span>
            <button 
              @click="removeCard(index)"
              class="text-red-500 hover:text-red-700"
              title="Karte löschen"
            >
              🗑️
            </button>
          </div>

          <div class="space-y-2">
            <input 
              v-model="card.front" 
              type="text"
              placeholder="Vorderseite (Begriff, Frage)"
              class="w-full px-3 py-2 border rounded"
              style="border-color: var(--color-border); background-color: var(--color-surface);"
            />
            <textarea 
              v-model="card.back" 
              rows="2"
              placeholder="Rückseite (Definition, Antwort)"
              class="w-full px-3 py-2 border rounded resize-none"
              style="border-color: var(--color-border); background-color: var(--color-surface);"
            />
            <input 
              v-model="card.tags" 
              type="text"
              placeholder="Tags (optional, kommasepariert)"
              class="w-full px-3 py-2 text-sm border rounded"
              style="border-color: var(--color-border); background-color: var(--color-surface);"
            />
          </div>
        </div>
      </div>

      <p v-if="formData.cards.length === 0" class="text-center py-8 text-gray-500">
        Noch keine Karteikarten. Klicke auf "+ Karte hinzufügen" um zu starten.
      </p>
    </div>
  </BaseLearningMethodForm>
</template>

<script setup lang="ts">
import { ref, onMounted } from 'vue'
import BaseLearningMethodForm from './BaseLearningMethodForm.vue'

const props = defineProps<{
  window: any
}>()

const formData = ref({
  cards: [{ front: '', back: '', tags: '' }]
})

onMounted(() => {
  // Lade existierende Daten beim Bearbeiten
  if (props.window.payload.instanceData?.config?.cards) {
    formData.value.cards = props.window.payload.instanceData.config.cards
  }
})

const addCard = () => {
  formData.value.cards.push({ front: '', back: '', tags: '' })
}

const removeCard = (index: number) => {
  if (formData.value.cards.length > 1) {
    formData.value.cards.splice(index, 1)
  }
}

const handleSave = (baseData: any) => {
  return {
    ...baseData,
    config: {
      ...baseData.config,
      cards: formData.value.cards.filter(c => c.front.trim() || c.back.trim())
    }
  }
}
</script>
```
