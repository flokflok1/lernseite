        # Direct repository calls, caching, fallback logic
        cache_key = f"CACHE:I18N:BUNDLE:{language_code}:{namespace}"
        cached = CacheService.cache_get(cache_key)

NEW MODULAR (translations.py):
```python
class TranslationManager:
    """Focused manager for translation CRUD"""
    BUNDLE_CACHE_TTL = 3600

    @staticmethod
    def get_bundle(language_code, namespace):
        # Uses PostgreSQL function, modern caching
        cache_key = f"i18n:bundle:{language_code}:{namespace}"
        cached = CacheService.cache_get(cache_key)
        if cached:
            return cached
        query = "SELECT get_i18n_bundle(%s::varchar, %s::varchar) AS bundle"
        result = fetch_one(query, (language_code, namespace))
        CacheService.cache_set(cache_key, result['bundle'], BUNDLE_CACHE_TTL)
        return result['bundle']
```

BRIDGE FACADE (bridge.py):
```python
class I18nService:
    """Backward-compatible facade delegating to new managers"""

    @staticmethod
    def get_bundle(language_code, namespace):
        return TranslationManager.get_bundle(language_code, namespace)

    @staticmethod
    def generate_ai_translation(key_id, target_language, user_id):
        primary_language = LanguageManager.get_primary_language()
        return AITranslationGenerator.generate_ai_translation(
            key_id, target_language, user_id, primary_language
        )
```
